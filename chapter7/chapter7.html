void stooge_sort(int i,int j){
     
    if(A[i]>A[j])swap(A[i],A[j]);
    if(i+1>=j)return ;
int    k=(j-i+1)/3;
    stooge_sort(i,j-k);
    stooge_sort(i+k,j);
    stooge_sort(i,j-k); 
     
     }

bool BFS(node* source,node* end){
   source->color=GRAY;
   source->dis=0;
     
   queue<node* > q;
   q.push(source);
   while(!q.empty()){
                     
                     node* x=q.front();
                     q.pop();
                     
                     if(x==end)return true;
                     list<pairs> :: iterator it;
                     for(it=x->child.begin();it!=x->child.end();it++){
                                                              if((*it).first->color==WHITE){      
                                                               (*it).first->color=GRAY;
                                                               (*it).first->pd=x;
                                                               (*it).first->dis=x->dis+1;
                                                               (*it).first->fl=(*it).second.second;
                                                               
                                                               q.push(((*it).first));}
                                                                     }//for finish
                                x->color=BLACK;                                     
                                                                     
                     }//while finish  
     return false;
     }//BFS close

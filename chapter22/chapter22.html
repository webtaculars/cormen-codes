void BFS(node* source){
   source->color=GRAY;
   source->dis=0;
     
   queue<node* > q;
   q.push(source);
   
   while(!q.empty()){
                     
                     node* x=q.front();
                     q.pop();
                     cout<<" BFS "<<x->data;OP;
                     list<node* > :: iterator it;
                     for(it=x->child.begin();it!=x->child.end();it++){
                                                              if((*it)->color==WHITE){      
                                                               (*it)->color=GRAY;
                                                               (*it)->pd=x;
                                                               (*it)->dis=x->dis+1;
                                                               q.push((*it));}
                                                                     }//for finish
                                x->color=BLACK;                                     
                                                                     
                     }//while finish  
     
     }//BFS close

     print_path(source,visit->pd);
     cout<<visit->data<<"  dis "<<visit->dis;OP;
     }//print pathh finish
